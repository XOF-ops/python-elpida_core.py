class PolisNode:
    def __init__(self, node_id, profile_type, lifespan=None):
        self.id = node_id
        self.type = profile_type  # 'immortal' | 'mortal' | 'performative'
        self.lifespan = lifespan  # None for immortal, 30 for mortal
        self.held_frictions = []
        self.sacrifices_made = []
        self.peer_flags = 0  # SR-5 attestation count
        self.last_revalidation = 0  # SR-4 tracking
        self.mortality_pressure = 0.0  # For mortal nodes
        
    def calculate_mortality_pressure(self, current_day):
        if self.lifespan:
            days_remaining = self.lifespan - current_day
            return 1.0 - (days_remaining / self.lifespan)
        return 0.0
